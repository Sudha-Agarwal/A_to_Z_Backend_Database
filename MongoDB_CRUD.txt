//Read

db.courses.find();

// Find documents with a specific condition
db.courses.find({ 'coursedetails.duration': { $gt: 10 } });

//update
// Update a single document
db.courses.updateOne(
  { name: 'MongoDB for Beginners' },
  { $set: { 'instructor.name': 'Updated Instructor' } }
);

// Update multiple documents
db.courses.updateMany(
  { 'coursedetails.duration': { $gt: 10 } },
  { $set: { 'instructor.name': 'Updated Instructor' } }
);


//delete

db.courses.deleteOne({ name: 'MongoDB for Beginners' });
